function calculateVolumeAndDensity() {
    const length = parseFloat(document.getElementById('length').value);
    const width = parseFloat(document.getElementById('width').value);
    const height = parseFloat(document.getElementById('height').value);
    const psa = parseFloat(document.getElementById('psa').value); // Optional PSA input

    const volumeResultEl = document.getElementById('prostate-volume-result');
    const densityResultEl = document.getElementById('psa-density-result');
    const errorEl = document.getElementById('calculator-error'); // Add an error display element

    // Clear previous results and errors
    volumeResultEl.textContent = '-- cc';
    densityResultEl.textContent = '--';
    if(errorEl) errorEl.textContent = '';


    // Basic Validation
    if (isNaN(length) || isNaN(width) || isNaN(height) || length <= 0 || width <= 0 || height <= 0) {
        if(errorEl) errorEl.textContent = 'Please enter valid positive numbers for all dimensions (Length, Width, Height).';
        else alert('Please enter valid positive numbers for all dimensions (Length, Width, Height).');
        return;
    }
    if (length > 20 || width > 20 || height > 20) { // Example upper limit
        if(errorEl) errorEl.textContent = 'One or more dimensions seem unusually large. Please double-check your inputs (expected in cm).';
        else alert('One or more dimensions seem unusually large. Please double-check your inputs (expected in cm).');
        // Potentially still calculate or offer to proceed
    }


    const volume = (length * width * height) * 0.52;
    volumeResultEl.textContent = volume.toFixed(2) + ' cc';

    if (!isNaN(psa) && psa > 0 && volume > 0) {
        const density = psa / volume;
        densityResultEl.textContent = density.toFixed(2);
    } else if (!isNaN(psa) && psa <= 0) {
        densityResultEl.textContent = 'PSA must be positive.';
    } else {
        densityResultEl.textContent = 'Enter PSA for density.';
    }
}

function clearResults() {
    document.getElementById('length').value = '';
    document.getElementById('width').value = '';
    document.getElementById('height').value = '';
    const psaInput = document.getElementById('psa');
    if (psaInput) psaInput.value = '';

    document.getElementById('prostate-volume-result').textContent = '-- cc';
    document.getElementById('psa-density-result').textContent = '--';
    const errorEl = document.getElementById('calculator-error');
    if(errorEl) errorEl.textContent = '';
}

III. Technical SEO Enhancements

    Core Web Vitals (CWV): Google uses these as ranking factors. Focus on:
        Largest Contentful Paint (LCP): How quickly the main content loads. Optimize images, defer non-critical CSS/JS, improve server response time.
        Interaction to Next Paint (INP): (Replaced First Input Delay - FID) How responsive your page is to user interactions. Optimize JavaScript, reduce main thread work.
        Cumulative Layout Shift (CLS): Visual stability of the page. Specify image dimensions, reserve space for ads/embeds.
        Tools: Google PageSpeed Insights, Lighthouse in Chrome DevTools, Google Search Console's Core Web Vitals report.
    Mobile-First Indexing: Your site must be perfectly responsive and offer an excellent mobile experience. Test on various devices.
    robots.txt Refinements:
        Ensure you're not accidentally disallowing important CSS, JS, or image folders that are necessary for rendering the page.
        Allow: /css/
        Allow: /js/
        Allow: /images/
    Meta Robots Tag Control (per page):
        While robots.txt gives directives for crawlers, meta robots tags give instructions for indexing specific pages.
        Default (no need to add if you want indexing and following): <meta name="robots" content="index, follow">
        Use noindex for utility pages you don't want in search results (e.g., internal thank-you pages, temporary pages). <meta name="robots" content="noindex, follow"> or <meta name="robots" content="noindex, nofollow">. Be very careful with this.
    Canonicalization:
        Use rel="canonical" tags to specify the preferred version of a page if duplicate or very similar content exists.
            E.g., if prostatevolumecalc.com/calculator.html and prostatevolumecalc.com/calculator.html?print=true exist, the version with the parameter should have a canonical tag pointing to prostatevolumecalc.com/calculator.html.
            <link rel="canonical" href="https://prostatevolumecalc.com/canonical-version-of-page.html" />
        Ensure consistent use of www vs. non-www and HTTP vs. HTTPS. Pick one and redirect others (301) to it. Replit often handles this.
    XML Sitemap:
        Include all indexable pages.
        Update it automatically or regularly when content changes.
        Submit it via Google Search Console.
        Consider including <lastmod>, <changefreq>, and <priority> tags (though Google often determines priority itself).
    Structured Data for Site Name & Sitelinks Search Box:
    HTML

    <script type="application/ld+json">
    {
      "@context": "https://schema.org",
      "@type": "WebSite",
      "name": "Prostate Volume Calculator & Health Hub",
      "url": "https://prostatevolumecalc.com/",
      "potentialAction": {
        "@type": "SearchAction",
        "target": "https://prostatevolumecalc.com/search?q={search_term_string}", // If you implement on-site search
        "query-input": "required name=search_term_string"
      }
    }
    </script>

    HTTPS: Essential for security, trust, and SEO. Replit provides this by default for hosted projects.
    Error Handling:
        Custom, user-friendly 404.html page that guides users back to relevant content (e.g., homepage, sitemap, search).
        Regularly check Google Search Console for crawl errors (404s, server errors) and fix them. Implement 301 redirects for moved/deleted pages to relevant new URLs.
